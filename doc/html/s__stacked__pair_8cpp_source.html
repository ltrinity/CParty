84pf_t s_stacked_pair::compute_energy_restricted (int i, int j, str_features *fres)
85// compute the free energy of the structure closed by this stacked pair
86{
87 if (fres[i].pair == j && fres[j].pair==i && !can_pair(sequence[i],sequence[j])){
88 return V->get_energy (i+1,j-1);
89 }else if (fres[i].pair == j && fres[j].pair==i &&
90 fres[i+1].pair == j-1 && fres[j-1].pair==i+1 &&
91 !can_pair(sequence[i+1],sequence[j-1])){
92 return V->get_energy (i+1,j-1);
93 }
94 else{
95 return compute_energy(i,j);
96 }
97}
