67void get_arguments (int argc, char *argv[])
68// PRE: None
69// POST: Get the parameters from the command line
70{
71 int c;
72 extern char *optarg;
73 extern int optind;
74 int errflag = 0;
75
76 while ((c = getopt (argc, argv, "t:m:s:r:k:n:p:dh?")) != -1)
77 {
78 switch (c)
79 {
80 case 't':
81 temperature = atof (optarg);
82 break;
83 case 'm':
84 if (strcmp (optarg, "DNA") == 0) dna_or_rna = DNA;
85 else if (strcmp (optarg, "RNA") == 0) dna_or_rna = RNA;
86 else errflag = 1;
87 break;
88 case 's':
89 strcpy (sequence, optarg);
90 break;
91 case 'r':
92 strcpy (restricted, optarg);
93 break;
94 case 'k':
95 strcpy (known_structure, optarg);
96 break;
97 case 'n':
98 num_subopt = atoi (optarg);
99 break;
100 case 'p':
101 strcpy (parameter_filename, optarg);
102 break;
103 case 'd':
104 debug = 1;
105 break;
106 case 'h':
107 case '?':
108 default:
109 errflag = 1;
110 }
111 }
112 if (errflag || optind != argc || argc < 3)
113 usage (argv[0]);
114
115}
