39bool validateStructure(char* structure, char* sequence){
40 // check lengths
41 if(strlen(structure) > MAXSLEN){
42 printf("Structure length greater than max length\n");

44 }
45
46 if(strlen(structure) <= 0){
47 printf("length of structure is <= 0. This shouldn't even be able to happen.\n");
48 return false;
49 }
50
51 //printf("strlen: %d %d\n",strlen(structure), strlen(sequence));
52 if(strlen(structure) != strlen(sequence)){
53 printf("Length of sequence and corresponding structure must have same length.\n");
54 return false;

57 //check if any characters other than ._(){}[]
58 for(char* it = structure; *it; ++it) {
59 char curr = *it;
60 // Structure must only contain ._(){}[
61 if (!(curr == '.' || curr == '_'
62 || curr == '(' || curr == '{' || curr == '['
63 || curr == ')' || curr == '}' || curr == ']'))
64 {
65 printf("Structure must only contain ._(){}[] \n");
66 return false;
67 }
68 }
69
70 std::string openBracketArray ("({[");
71 std::string closeBracketArray (")}]");
72 std::stack<char> mystack;
73
74 for(int i=0; i<strlen(structure);i++) {
75 char curr = structure[i];
76 //printf("d: %c\n",structure[i]);
77 if(curr == '(' || curr == '{' || curr == '[') { //is open bracket
78 mystack.push(structure[i]);
79 } else
80 if(curr == ')' || curr == '}' || curr == ']') { //is close bracket
81
82 // if stack is empty that means there are more right brackets than left brackets
83 if (mystack.empty() == true ) {
84 printf("Structure is invalid: more right parentheses than left parentheses\n");
85 return false;
86 }
87
88 if(closeBracketArray.find_first_of(curr) != openBracketArray.find_first_of(mystack.top())){ //if current bracket is not corresponding bracket of what we popped
89 printf("Structure bracket types must match and be pseudoknot free\n");
90 return false;
91 }
92
93 mystack.pop();
94 }
95 }
96
97 if(mystack.empty() == false){
98 printf("Structure is invalid: more left parentheses than right parentheses\n");
99 return false;
100 }
101
102 return true;
103
104}

109 return false;

115
